# Task View Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a Tasks tab to the admin dashboard with a two-column layout — task list on the left, inline tool call logs for the selected task on the right.

**Architecture:** All changes are confined to one file: `remote-gateway/core/admin_dashboard.html`. The tab follows the existing `switchTab()` / `view-*` pattern. The right panel populates the existing global log maps (`_logsBodyMap`, `_logsToolMap`, etc.) so the existing `openLogDrawer()` works without modification. No new backend endpoints are needed — `GET /admin/api/tasks` and `GET /admin/api/logs?task_id=<id>` already exist.

**Tech Stack:** Vanilla HTML, CSS, JavaScript — no build step, no external dependencies beyond what is already in the file.

- [ ] **Step 1: Add fmtDuration() helper**

Find the `fmtDate()` function (around line 935). Insert `fmtDuration()` immediately after `fmtDate()`:

```js
  function fmtDuration(createdAt, completedAt) {
    if (!completedAt) return '—';
    const ms = (completedAt - createdAt) * 1000;
    if (ms < 1000) return Math.round(ms) + ' ms';
    if (ms < 60000) return (ms / 1000).toFixed(1) + ' s';
    return Math.round(ms / 60000) + ' min';
  }
```

Note: `created_at` and `completed_at` from the API are Unix timestamps (seconds as floats).

- [ ] **Step 3: Add loadTasks() function**

Add the following after the `renderLogsTable()` function (around line 1804, after its closing `}`):

```js
  // ----------------------------------------------------------------
  // Tasks
  // ----------------------------------------------------------------
  async function loadTasks() {
    const status = document.getElementById('tasks-filter-status').value;
    let url = apiUrl('/api/tasks') + '&limit=100';
    if (status) url += '&status=' + encodeURIComponent(status);
    try {
      const res  = await fetch(url);
      const data = await res.json();
      renderTasksTable(data.tasks || []);
    } catch (e) {
      document.getElementById('tasks-tbody').innerHTML =
        '<tr><td colspan="5" style="color:var(--orange);font-style:italic;text-align:center;">Failed to load tasks.</td></tr>';
      console.error('loadTasks error:', e);
    }
  }

  function renderTasksTable(tasks) {
    _selectedTaskId = null;
    _tasksMap = {};
    document.getElementById('task-logs-placeholder').style.display = 'block';
    document.getElementById('task-logs-content').style.display = 'none';

    const tbody = document.getElementById('tasks-tbody');
    if (!tasks.length) {
      tbody.innerHTML = '<tr><td colspan="5" style="color:var(--text-muted);font-style:italic;text-align:center;">No tasks found.</td></tr>';
      return;
    }
    tasks.forEach(t => { _tasksMap[t.task_id] = t; });
    tbody.innerHTML = tasks.map(t => {
      const goalText = t.goal || '—';
      const truncGoal = goalText.length > 60 ? goalText.slice(0, 60) + '…' : goalText;
      const badge = t.status === 'active'
        ? '<span class="task-badge-active">active</span>'
        : '<span class="task-badge-completed">completed</span>';
      const created = t.created_at ? fmtDate(new Date(t.created_at * 1000).toISOString()) : '—';
      const duration = fmtDuration(t.created_at, t.completed_at);
      return '<tr onclick="selectTask(\'' + escJs(t.task_id) + '\')" data-task-id="' + escHtml(t.task_id) + '">'
        + '<td title="' + escHtml(goalText) + '">' + escHtml(truncGoal) + '</td>'
        + '<td style="font-size:0.78rem;">' + escHtml(t.user_id || '—') + '</td>'
        + '<td>' + badge + '</td>'
        + '<td style="font-size:0.75rem;white-space:nowrap;">' + escHtml(created) + '</td>'
        + '<td>' + escHtml(duration) + '</td>'
        + '</tr>';
    }).join('');
  }
```

```js
  function selectTask(taskId) {
    const task = _tasksMap[taskId];
    if (!task) return;
    _selectedTaskId = taskId;
    document.querySelectorAll('#tasks-tbody tr').forEach(r => {
      r.classList.toggle('selected', r.dataset.taskId === taskId);
    });
    document.getElementById('task-logs-goal').textContent = task.goal || '—';
    const outcomeEl = document.getElementById('task-logs-outcome');
    if (task.outcome) {
      outcomeEl.textContent = task.outcome;
      outcomeEl.style.display = 'block';
    } else {
      outcomeEl.style.display = 'none';
    }
    document.getElementById('task-logs-placeholder').style.display = 'none';
    document.getElementById('task-logs-content').style.display = 'block';
    loadTaskLogs(taskId);
  }
```

- [ ] **Step 2: Add loadTaskLogs() and renderTaskLogsPanel()**

Add immediately after `selectTask()`:

```js
  async function loadTaskLogs(taskId) {
    const url = apiUrl('/api/logs') + '&task_id=' + encodeURIComponent(taskId) + '&limit=100';
    try {
      const res  = await fetch(url);
      const logs = await res.json();
      renderTaskLogsPanel(Array.isArray(logs) ? logs : []);
    } catch (e) {
      document.getElementById('task-logs-tbody').innerHTML =
        '<tr><td colspan="4" style="color:var(--orange);font-style:italic;text-align:center;">Failed to load tool calls.</td></tr>';
      console.error('loadTaskLogs error:', e);
    }
  }

  function renderTaskLogsPanel(logs) {
    // Populate global log maps so openLogDrawer() works for rows in this panel.
    logs.forEach(row => {
      if (row.input_body       != null) _logsBodyMap[row.id]         = row.input_body;
      if (row.error_message    != null) _logsErrorMap[row.id]        = row.error_message;
      if (row.response_preview != null) _logsResponseMap[row.id]     = row.response_preview;
      if (row.response_size    != null) _logsResponseSizeMap[row.id] = row.response_size;
      _logsToolMap[row.id] = row.tool_name || '';
      _logsTimeMap[row.id] = row.called_at || '';
    });

    const tbody = document.getElementById('task-logs-tbody');
    if (!logs.length) {
      tbody.innerHTML = '<tr><td colspan="4" style="color:var(--text-muted);font-style:italic;text-align:center;">No tool calls recorded for this task.</td></tr>';
      return;
    }
    tbody.innerHTML = logs.map(row => {
      const statusBadge = row.success
        ? '<span class="badge badge-green">OK</span>'
        : row.error_type === 'PermissionError'
          ? '<span class="badge badge-red" title="Tool blocked by permissions">BLOCKED</span>'
          : '<span class="badge badge-orange" title="' + escHtml(row.error_type || '') + '">' + escHtml(row.error_type || 'ERR') + '</span>';
      const hasDetail = row.input_body != null || row.error_message != null || row.response_preview != null;
      const rowStyle  = hasDetail ? 'cursor:pointer;' : '';
      const onclick   = hasDetail ? ' onclick="toggleLogDrawer(' + escJs(String(row.id)) + ')"' : '';
      return '<tr style="' + rowStyle + '" data-log-id="' + escHtml(String(row.id)) + '"' + onclick + '>'
        + '<td style="white-space:nowrap;font-size:0.75rem;">' + escHtml(row.called_at || '—') + '</td>'
        + '<td>' + escHtml(row.tool_name) + '</td>'
        + '<td>' + fmtMs(row.duration_ms) + '</td>'
        + '<td>' + statusBadge + '</td>'
        + '</tr>';
    }).join('');
  }
```
